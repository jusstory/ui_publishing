custom alert

타이틀 있을 때 :
TNUI.module.alertui({title : 'tit',msg : 'aaa'},function(){ /* 콜백함수 */});
타이틀 있을 때 :
TNUI.module.alertui({msg : 'aaa'},function(){ /* 콜백함수 */});
타이틀 있을 때, 없을 때 모두 alertui 호출
타이틀이 있으면 title 값 입력, 없으면 입력X 

<button class="button" id="aa">alert 창 - 타이틀 있는 것</button>
<button class="button" id="bb">alert 창 - 타이틀 없는 것</button>

<script>
	var _call_back = function(){
		//callback 내용
		console.log(1);
	} 
	$('#aa').on('click', function(){//#aa를 눌렀을 때 
		TNUI.module.alertui({title : 'tit',msg : 'aaa'},function(){ 
			_call_back();
		});
	})
	$('#aa').on('click', function(){//#bb를 눌렀을 때 
		TNUI.module.alertui({msg : 'aaa'},function(){ 
			_call_back();
		});
	})
</script>
